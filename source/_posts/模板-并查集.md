```cpp
int ufs[Maxn];//UnionFindSet

inline void Init(int n){//初始化
    for (int i = 1;i<=n;i++){ufs[i] = i;}
}

int Find(int x){//查找
    return ufs[x]==x?x:ufs[x] = find(ufs[x]);
}

void Union(int x, int y){//合并
    int fx = find(x), fy = find(y);
    if(fx != fy){ ufs[fx] = fy; }
}
```
